1. git init - создать локальный git репозиторий
2. mkdir "имя папки" - создать папку
3. touch "имя файла.html" - создать файл
4. ls - просмотр содержимого директории
5. git status - просмотр состояния файлов под контролем git
6. git add "имя файла.js"/git add . - добавить один файл/добавить все файлы под контроль git
7. git commit -m "Текст коммита" - фиксируем и описываем изменения в файлах
